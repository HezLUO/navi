# Along Memory v2 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Implement Memory v2 so Along creates sourced episodes, consolidates reviewable long-term memory, recalls a small cross-session/cross-project packet, and gives the user transparent review controls.

**Architecture:** Keep the existing local-first TypeScript core and add focused memory modules instead of introducing a database. The runtime writes raw sessions, episodes, consolidation candidates, graph links, and recall packets; the provider consumes only a compact recall packet so Along keeps continuity without loading all memory.

**Tech Stack:** Node 22, TypeScript, Vitest, Express, React/Vite, local JSON and Markdown files.

Responsibility boundaries:

- `src/core/types.ts`: Memory v2 domain types, mode ordering, candidate/review/recall contracts, and graph schema unions.
- `src/core/paths.ts`: all project/global memory paths, including episodes, review, recall, procedures, and settings.
- `src/core/memory-store.ts`: readable file initialization plus read/write methods for episodes, journal append, settings, semantic memory, review inbox, procedures, forget ledger, and recall packets.
- `src/core/consolidation.ts`: deterministic consolidation from an episode into memory candidates and graph updates.
- `src/core/recall.ts`: budgeted recall packet generation with scope, status, intimacy, expiry, and review filters.
- `src/core/graph-store.ts`: required graph persistence with Memory v2 node/edge metadata.
- `src/core/runtime.ts`: session lifecycle integration: start uses recall, wrap-up creates episode and consolidation.
- `src/core/model-provider.ts`: deterministic provider uses recall hints without turning them into a memory dump.
- `src/cli/memory-review.ts`: pure CLI formatting for review inboxes.
- `src/cli/index.ts`: command dispatch for `along start` and `along memory review --list`.
- `src/server/app.ts`: expose richer wrap-up result for review visibility.
- `src/web/App.tsx` and `src/web/styles.css`: show a compact "What I remembered today" wrap-up result.

```ts
import type { EpisodeRecord, GraphEdge, GraphNode, MemoryCandidate, SessionEvent } from "./types";

export interface ConsolidationResult {
  candidates: MemoryCandidate[];
  graph: {
    nodes: GraphNode[];
    edges: GraphEdge[];
  };
}

export function consolidateEpisode(episode: EpisodeRecord, now = new Date().toISOString()): ConsolidationResult {
  const candidates = episode.events.flatMap((event) => candidateFromEvent(episode, event, now));
  const graphNodes: GraphNode[] = [
    {
      id: episode.id,
      type: "episode",
      label: episode.id,
      properties: { sessionId: episode.sessionId, repoPath: episode.repoPath },
      createdAt: now,
      metadata: baseMetadata(episode, now, "session", "minimal", "auto_approved"),
    },
    ...candidates.map(candidateToNode),
  ];
  const graphEdges: GraphEdge[] = [
    {
      id: `edge:${episode.sessionId}:created:${episode.id}`,
      from: `session:${episode.sessionId}`,
      to: episode.id,
      type: "session_created_episode",
      createdAt: now,
      metadata: baseMetadata(episode, now, "session", "minimal", "auto_approved"),
    },
    ...candidates.map((candidate) => candidateToEpisodeEdge(candidate, episode, now)),
  ];
  return { candidates, graph: { nodes: graphNodes, edges: graphEdges } };
}

function candidateFromEvent(episode: EpisodeRecord, event: SessionEvent, now: string): MemoryCandidate[] {
  if (event.type === "companion_observed") {
    return [candidate(episode, event.text, "semantic_fact", "project", "minimal", "auto_approved", now, event.projectArea)];
  }
  if (event.type === "user_corrected") {
    return [candidate(episode, event.text, "correction", "project", "minimal", "auto_approved", now, event.projectArea)];
  }
  if (event.type === "continuity_noted") {
    return [candidate(episode, event.text, "continuity", "global", "close", "pending_review", now, event.projectArea, "Guide companion tone only after user approval.")];
  }
  if (event.type === "procedure_suggested") {
    return [candidate(episode, event.text, "procedure", "global", "balanced", "pending_review", now, event.projectArea, "Suggest as a reusable procedure only after review.")];
  }
  if (event.type === "decision_recorded") {
    return [candidate(episode, event.text, "graph_relation", "project", "minimal", "auto_approved", now, event.projectArea)];
  }
  return [];
}

function candidate(
  episode: EpisodeRecord,
  text: string,
  type: MemoryCandidate["type"],
  scope: MemoryCandidate["scope"],
  intimacy: MemoryCandidate["memoryIntimacy"],
  reviewStatus: MemoryCandidate["reviewStatus"],
  now: string,
  projectArea?: string,
  allowedUse?: string,
): MemoryCandidate {
  return {
    id: `${type}:${episode.id}:${slug(text)}`,
    type,
    text,
    sourceEpisodeIds: [episode.id],
    confidence: "explicit",
    scope,
    memoryIntimacy: intimacy,
    reviewStatus,
    status: "active",
    visibility: "user_visible",
    createdAt: now,
    updatedAt: now,
    targetPath: projectArea,
    allowedUse,
    useAs: scope === "global" && type === "semantic_fact" ? "hint_not_fact" : "fact",
  };
}

function baseMetadata(
  episode: EpisodeRecord,
  now: string,
  scope: MemoryCandidate["scope"],
  memoryIntimacy: MemoryCandidate["memoryIntimacy"],
  reviewStatus: MemoryCandidate["reviewStatus"],
) {
  return {
    scope,
    sourceEpisodeIds: [episode.id],
    confidence: "explicit" as const,
    memoryIntimacy,
    status: "active" as const,
    visibility: "user_visible" as const,
    reviewStatus,
    createdAt: now,
    updatedAt: now,
  };
}

function candidateToNode(candidate: MemoryCandidate): GraphNode {
  const typeByCandidate = {
    semantic_fact: "learned_fact",
    correction: "correction",
    graph_relation: "decision",
    continuity: "continuity_note",
    procedure: "procedure_candidate",
  } as const;
  return {
    id: candidate.id,
    type: typeByCandidate[candidate.type],
    label: candidate.text,
    properties: { targetPath: candidate.targetPath ?? "" },
    createdAt: candidate.createdAt,
    metadata: candidate,
  };
}

function candidateToEpisodeEdge(candidate: MemoryCandidate, episode: EpisodeRecord, now: string): GraphEdge {
  const edgeTypeByCandidate = {
    semantic_fact: "episode_supported_fact",
    correction: "episode_recorded_correction",
    graph_relation: "decision_supersedes_assumption",
    continuity: "episode_created_continuity_note",
    procedure: "procedure_candidate_derived_from_episode",
  } as const;
  return {
    id: `edge:${episode.id}:${candidate.id}`,
    from: episode.id,
    to: candidate.id,
    type: edgeTypeByCandidate[candidate.type],
    createdAt: now,
    metadata: candidate,
  };
}

function slug(text: string): string {
  return text.toLowerCase().replace(/[^a-z0-9]+/g, "-").replace(/(^-|-$)/g, "").slice(0, 48) || "memory";
}
```

```ts
  const corrections = candidates.filter((candidate) => candidate.type === "correction" && candidate.scope === "project" && candidate.reviewStatus === "auto_approved");

  await this.appendUniqueJson(path.join(this.globalDir, "review", "inbox.json"), pending.filter((candidate) => candidate.scope === "global"));

private async appendUniqueJson<T extends { id: string }>(filePath: string, items: T[]): Promise<void> {
  const existing = await this.readJsonFile<T[]>(filePath, []);
  const byId = new Map(existing.map((item) => [item.id, item]));
  for (const item of items) byId.set(item.id, item);
  await this.writeJsonFile(filePath, [...byId.values()]);
}

private reviewReason(candidate: MemoryCandidate): string {
  if (candidate.type === "continuity") return "close continuity requires review";
  if (candidate.type === "procedure") return "procedure candidates require review before use";
  if (candidate.scope === "global") return "cross-project memory requires review";
  return "memory requires review";
}
```

```ts
import { memoryModes, type CuriosityItem, type MemoryCandidate, type MemoryMode, type ProcedureCard, type RecallItem, type RecallPacket } from "./types";

export interface BuildRecallInput {
  sessionId: string;
  repoPath: string;
  now: string;
  mode: MemoryMode;
  activeCuriosity?: CuriosityItem;
  projectFacts: MemoryCandidate[];
  projectCorrections: MemoryCandidate[];
  globalPreferences: MemoryCandidate[];
  globalContinuity: MemoryCandidate[];
  crossProjectPatterns: MemoryCandidate[];
  procedures: ProcedureCard[];
}

export function buildRecallPacket(input: BuildRecallInput): RecallPacket {
  const blocked: Record<string, number> = {};
  const allow = <T extends MemoryCandidate>(items: T[]) =>
    items.filter((item) => {
      const reason = blockedReason(item, input.mode, input.now);
      if (reason) {
        blocked[reason] = (blocked[reason] ?? 0) + 1;
        return false;
      }
      return true;
    });

  const stableFacts = allow(input.projectFacts).slice(0, 3).map(toRecallItem);
  const activeCorrections = allow(input.projectCorrections).slice(0, 3).map(toRecallItem);
  const preferences = allow(input.globalPreferences).slice(0, 2).map(toRecallItem);
  const companionContinuity = allow(input.globalContinuity).slice(0, 2).map(toRecallItem);
  const possiblyRelevantPatterns = allow(input.crossProjectPatterns).slice(0, 2).map((item) => toRecallItem({ ...item, useAs: "hint_not_fact" }));
  const activeProcedures = allow(input.procedures)
    .filter((procedure) => procedure.lifecycle === "active")
    .slice(0, 2)
    .map((procedure) => ({ id: procedure.id, title: procedure.title, trigger: procedure.trigger }));

  const provenance = [
    ...stableFacts,
    ...activeCorrections,
    ...preferences,
    ...companionContinuity,
    ...possiblyRelevantPatterns,
  ].map((item) => item.source);

  return {
    sessionId: input.sessionId,
    repoPath: input.repoPath,
    generatedAt: input.now,
    mode: input.mode,
    projectRecall: {
      activeCuriosity: input.activeCuriosity ? { id: input.activeCuriosity.id, question: input.activeCuriosity.question } : undefined,
      stableFacts,
      activeCorrections,
    },
    globalRecall: {
      preferences,
      companionContinuity,
      possiblyRelevantPatterns,
    },
    proceduralRecall: {
      activeProcedures,
    },
    blockedMemories: Object.entries(blocked).map(([reason, count]) => ({ reason, count })),
    provenance: [...new Set(provenance)],
  };
}

function blockedReason(item: MemoryCandidate, mode: MemoryMode, now: string): string | undefined {
  if (item.status !== "active") return "inactive";
  if (item.reviewStatus === "pending_review" || item.reviewStatus === "rejected") return "requires_review";
  if (item.expiresAt && item.expiresAt < now) return "expired";
  if (memoryModes.indexOf(item.memoryIntimacy) > memoryModes.indexOf(mode)) return "over_mode";
  return undefined;
}

function toRecallItem(item: MemoryCandidate): RecallItem {
  return {
    id: item.id,
    text: item.text,
    source: item.sourceEpisodeIds[0] ?? item.id,
    useAs: item.useAs,
  };
}
```

```ts
import fs from "node:fs/promises";
import path from "node:path";

import { getGlobalAlongDir, getProjectAlongDir } from "../core/paths";
import type { ReviewInboxItem } from "../core/types";

export function formatReviewInbox(items: ReviewInboxItem[]): string {
  if (items.length === 0) return "No pending memory review items.\n";
  return `${items.map((item) => [
    `- ${item.id}`,
    `  type: ${item.type}`,
    `  scope: ${item.scope}`,
    `  intimacy: ${item.memoryIntimacy}`,
    `  review: ${item.reviewStatus}`,
    `  source: ${item.sourceEpisodeIds.join(", ")}`,
    `  text: ${item.text}`,
    `  reason: ${item.reason}`,
  ].join("\n")).join("\n\n")}\n`;
}

export async function loadReviewInbox(repoPath = process.cwd(), homeDir = os.homedir()): Promise<ReviewInboxItem[]> {
  const projectItems = await readInbox(path.join(getProjectAlongDir(repoPath), "review", "inbox.json"));
  const globalItems = await readInbox(path.join(getGlobalAlongDir(homeDir), "review", "inbox.json"));
  return [...projectItems, ...globalItems];
}

async function readInbox(filePath: string): Promise<ReviewInboxItem[]> {
  try {
    return JSON.parse(await fs.readFile(filePath, "utf8")) as ReviewInboxItem[];
  } catch {
    return [];
  }
}
```
